    ```ts
    import { open } from '@tauri-apps/plugin-dialog';

    const STORE_FILE = 'preferences.json';
    const STORAGE_PATH_KEY = 'storageFolderPath';

    async function getStore() {
      return load(STORE_FILE, { autoSave: false });
    }

    export async function loadStoragePath(): Promise<string | null> {
      const store = await getStore();
      const value = await store.get<string>(STORAGE_PATH_KEY);
      return value ?? null;
    }

    export async function pickAndSaveStoragePath(): Promise<string | null> {
      const folder = await open({ directory: true, multiple: false });
      if (!folder || typeof folder !== 'string') return null;
      const store = await getStore();
      await store.set(STORAGE_PATH_KEY, folder);
      await store.save();
      return folder;
    }

    export async function clearStoragePath(): Promise<void> {
      const store = await getStore();
      await store.delete(STORAGE_PATH_KEY);
      await store.save();
    }
    ```

    Implement `src/lib/db.ts`:
    ```ts
    import Database from '@tauri-apps/plugin-sql';

    export class DatabaseInitError extends Error {
      constructor(message: string, public readonly cause?: unknown) {
        super(message);
        this.name = 'DatabaseInitError';
      }
    }

    const instances = new Map<string, Database>();

    function buildConnectionString(storageFolderPath: string): string {
      const normalized = storageFolderPath.replace(/\\/g, '/').replace(/\/$/, '');
      return `sqlite:${normalized}/bili-mushroom.db`;
    }

    export async function getDatabase(storageFolderPath: string): Promise<Database> {
      const cs = buildConnectionString(storageFolderPath);
      const cached = instances.get(cs);
      if (cached) return cached;
      try {
        const db = await Database.load(cs);
        instances.set(cs, db);
        return db;
      } catch (err) {
        throw new DatabaseInitError('Failed to open database', err);
      }
    }

    export async function verifyWalMode(db: Database): Promise<boolean> {
      const rows = await db.select<{ journal_mode: string }[]>('PRAGMA journal_mode', []);
      if (!Array.isArray(rows) || rows.length === 0) return false;
      return String(rows[0].journal_mode).toLowerCase() === 'wal';
    }

    export async function initializeDatabase(storageFolderPath: string): Promise<void> {
      const db = await getDatabase(storageFolderPath);
      try {
        // Pitfall 1 safeguard: run PRAGMA outside the migration transaction
        await db.execute('PRAGMA journal_mode=WAL', []);
      } catch (err) {
        throw new DatabaseInitError('Failed to set WAL mode', err);
      }
      const walActive = await verifyWalMode(db);
      if (!walActive) {
        throw new DatabaseInitError('WAL mode could not be enabled');
      }
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User OS folder picker → JS string → `Database.load(sqlite:${abs})` | User-chosen path becomes a SQLite connection string; path traversal / injection surface |
| `app_metadata` table | First persistent data surface — any SQL written here must use parameterized queries |
| `preferences.json` → app data dir | Plaintext JSON read/written by tauri-plugin-store |

## STRIDE Threat Register (ASVS L1)

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-06 | Tampering | SQL migration file (`0001_initial.sql`) | mitigate | Checked into git; reviewed on each PR; hash-stable via Cargo build (include_str!) |
| T-01-07 | Injection | `Database.load(sqlite:${path})` where path comes from user folder picker | mitigate | Path only travels through native OS folder picker (not typed) — picker returns a filesystem-valid absolute path. `db.ts` normalizes separators and strips trailing slash but does NOT rewrite the path. All subsequent queries (`app_metadata`) are written as static SQL with `[]` parameters — no interpolation |
| T-01-08 | Information Disclosure | `preferences.json` contains absolute path to user library folder | accept | Low sensitivity; stored only in user-owned app data dir; no secrets. Path is not PII per se |
| T-01-09 | DoS | Migration transaction leaves WAL PRAGMA ineffective (Pitfall 1) | mitigate | Belt-and-suspenders: `db.ts` runs PRAGMA again outside transaction AND calls `verifyWalMode` — throws `DatabaseInitError` if WAL is not active, preventing app from starting with a degraded-durability DB |
| T-01-10 | Tampering | Connection string mismatch (Pitfall 2 / A2) causes migrations to silently not apply | mitigate | Comment added in `lib.rs` marking this as a runtime-verified assumption; Plan 03 smoke checkpoint confirms `app_metadata` table exists in the created DB file. If it fails, a gap-closure plan switches to Pattern 4 (custom Rust init command) |
| T-01-11 | Elevation of Privilege | Over-granted Tauri ACL permissions | mitigate | `capabilities/default.json` only lists exact permissions used: `sql:allow-execute/select/load/close`, `dialog:allow-open` (no save), `store:allow-get/set/save/load` — no wildcard `fs:*`, no shell, no process spawn |
</threat_model>
