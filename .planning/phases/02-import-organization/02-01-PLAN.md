```rust
// src-tauri/src/commands/exif.rs
#[derive(serde::Serialize)]
pub struct ExifData {
    pub date: Option<String>,  // ISO "YYYY-MM-DD" (already normalized from "YYYY:MM:DD HH:MM:SS")
    pub lat: Option<f64>,      // decimal degrees, signed (S/W negative)
    pub lng: Option<f64>,
}

#[tauri::command]
pub async fn parse_exif(path: String) -> Result<ExifData, String>;

// src-tauri/src/commands/import.rs
#[derive(serde::Deserialize)]
pub struct ImportPayload {
    pub source_path: String,        // absolute path to source file
    pub original_filename: String,
    pub species_name: String,
    pub date_found: String,         // ISO YYYY-MM-DD
    pub country: String,
    pub region: String,
    pub lat: Option<f64>,
    pub lng: Option<f64>,
    pub notes: String,
}

#[derive(serde::Serialize)]
pub struct FindRecord {
    pub id: i64,
    pub photo_path: String,          // relative to storage_root
    pub original_filename: String,
    pub species_name: String,
    pub date_found: String,
    pub country: String,
    pub region: String,
    pub lat: Option<f64>,
    pub lng: Option<f64>,
    pub notes: String,
    pub created_at: String,
}

#[derive(serde::Serialize)]
pub struct ImportSummary {
    pub imported: Vec<FindRecord>,
    pub skipped: Vec<String>,  // original_filename list
}

#[derive(serde::Serialize, Clone)]
pub struct ImportProgress {
    pub current: usize,
    pub total: usize,
    pub filename: String,
}

#[tauri::command]
pub async fn import_find(
    app: tauri::AppHandle,
    storage_path: String,
    payloads: Vec<ImportPayload>,
) -> Result<ImportSummary, String>;

#[tauri::command]
pub async fn get_finds(storage_path: String) -> Result<Vec<FindRecord>, String>;
```

Event channel: `import-progress` with `ImportProgress` payload, emitted after each file is processed (success OR skip).

DB access note: `import_find` and `get_finds` MUST open the SQLite DB using the same absolute path scheme the frontend uses (`<storage_path>/bili-mushroom.db`) via `rusqlite` (already pulled in transitively by libsqlite3-sys). Do NOT use tauri-plugin-sql from Rust — the plugin's migration runner is the only piece we rely on from the plugin; direct queries use `rusqlite::Connection::open`. This keeps the Rust command self-contained and avoids the plugin's JS-only query API. Add `rusqlite = { version = "0.31", features = ["bundled"] }` if not already transitively available — VERIFY during Task 1 and add explicitly to Cargo.toml if needed.
